# CMakeLists.txt для сборки llama-server как shared library (.so)
# Фокус на /v1/chat/completions

set(TARGET llama-server-so)

include_directories(${CMAKE_CURRENT_SOURCE_DIR} ${CMAKE_CURRENT_BINARY_DIR})

if (MINGW)
    add_compile_definitions(_WIN32_WINNT=${GGML_WIN_VER})
endif()

if (NOT LLAMA_HTTPLIB)
    message(FATAL_ERROR "LLAMA_HTTPLIB is OFF, cannot build llama-server. Hint: to skip building server, set -DLLAMA_BUILD_SERVER=OFF")
endif()

# Определяем флаг для исключения main()
add_compile_definitions(LLAMA_SERVER_LIBRARY=1)

set(TARGET_SRCS
    llama_server_lib.cpp
    # server.cpp включен в llama_server_lib.cpp через #include
    utils.hpp
)

# server.cpp будет скомпилирован с LLAMA_SERVER_LIBRARY=1, что исключит main()

set(PUBLIC_ASSETS
    index.html.gz
    loading.html
)

foreach(asset ${PUBLIC_ASSETS})
    set(input "${CMAKE_CURRENT_SOURCE_DIR}/public/${asset}")
    set(output "${CMAKE_CURRENT_BINARY_DIR}/${asset}.hpp")
    list(APPEND TARGET_SRCS ${output})
    add_custom_command(
        DEPENDS "${input}"
        OUTPUT "${output}"
        COMMAND "${CMAKE_COMMAND}" "-DINPUT=${input}" "-DOUTPUT=${output}" -P "${PROJECT_SOURCE_DIR}/scripts/xxd.cmake"
    )
    set_source_files_properties(${output} PROPERTIES GENERATED TRUE)
endforeach()

# Build as shared library
add_library(${TARGET} SHARED ${TARGET_SRCS})

# Set output name
if (UNIX AND NOT APPLE)
    set_target_properties(${TARGET} PROPERTIES OUTPUT_NAME "llama-server")
    set_target_properties(${TARGET} PROPERTIES SUFFIX ".so")
elseif (APPLE)
    set_target_properties(${TARGET} PROPERTIES OUTPUT_NAME "llama-server")
    set_target_properties(${TARGET} PROPERTIES SUFFIX ".dylib")
elseif (WIN32)
    set_target_properties(${TARGET} PROPERTIES OUTPUT_NAME "llama-server")
    set_target_properties(${TARGET} PROPERTIES SUFFIX ".dll")
endif()

# Install
install(TARGETS ${TARGET} LIBRARY DESTINATION lib)
install(FILES llama_server_api.h DESTINATION include)

target_include_directories(${TARGET} PRIVATE ../mtmd)
target_include_directories(${TARGET} PRIVATE ${CMAKE_SOURCE_DIR})
target_include_directories(${TARGET} PUBLIC 
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    $<INSTALL_INTERFACE:include>
)

target_link_libraries(${TARGET} PRIVATE common mtmd cpp-httplib ${CMAKE_THREAD_LIBS_INIT})

if (ANDROID)
    target_link_libraries(${TARGET} PRIVATE log)
endif()

if (WIN32)
    target_link_libraries(${TARGET} PRIVATE ws2_32)
endif()

target_compile_features(${TARGET} PRIVATE cxx_std_17)

# Export symbols
if (UNIX AND NOT APPLE)
    set_target_properties(${TARGET} PROPERTIES
        CXX_VISIBILITY_PRESET hidden
        VISIBILITY_INLINES_HIDDEN ON
    )
    target_compile_options(${TARGET} PRIVATE -fvisibility=hidden)
    # Экспортируем только функции из llama_server_api.h
    target_compile_definitions(${TARGET} PRIVATE LLAMA_SERVER_API_EXPORTS)
endif()

if (WIN32)
    target_compile_definitions(${TARGET} PRIVATE LLAMA_SERVER_API_EXPORTS)
endif()
